# 300. Longest Increasing Subsequence

**Space complexity**: $O(1)$
**time complexity**: $O(n)$

```cpp
class Solution {
public:
    int lengthOfLIS(vector<int>& nums)
    {
        int n = nums.size();
        vector<int> memo(n, 1);

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {

                if (nums[j] < nums[i] && memo[j] <= memo[i]) {
                    memo[i] = max(memo[i], memo[j]+1);
                }
            }
        }

        return *max_element(memo.begin(), memo.end());
    }
};
```
