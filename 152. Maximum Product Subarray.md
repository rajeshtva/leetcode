# 152. Maximum Product Subarray

[link](https://leetcode.com/problems/maximum-product-subarray/)
**Level**: Medium

**Memory Use**: MB

**Time**: ms

**Explanation**:

- brute force approach implemented usign recursion.

**Space complexity**: $O(1)$
**time complexity**: $O(n^2)$

```cpp
class Solution {
public:
    int maxProduct(vector<int>& nums)
    {
        int n = nums.size();

        int ans = INT_MIN;

        for (int i = 0; i < n; i++) {
            int temp;
            for (int j = i; j < n; j++) {
                if (i == j) {
                    temp = nums[i];
                } else {
                    temp = temp * nums[j];
                }
                ans = max(ans, temp);
            }
        }

        return ans;
    }
};

```

## using dynamic programming.(using simple variables)

```cpp
class Solution {
public:
    int maxProduct(vector<int>& nums)
    {
        int maxCurrent = 1;
        int minCurrent = 1;
        int temp = 1;
        int res = INT_MIN;

        for (int a : nums) {
            if (a == 0) {
                maxCurrent = 1;
                minCurrent = 1;
                res = max(res, 0);
            } else {
                temp = maxCurrent;
                maxCurrent = max({ temp * a, minCurrent * a, a });
                minCurrent = min({ temp * a, minCurrent * a, a });

                res = max(maxCurrent, res);
            }
        }

        return res;
    }
};
```
