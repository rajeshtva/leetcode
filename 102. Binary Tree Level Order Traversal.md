**Level**: Easy

**Memory Use**: 19.8 MB

**Time**: 36 ms

**Explanation**:
- used BFS to iterate over all nodes. 

- **Space complexity**: $O(1)$
  **time complexity**: $O(n.log(k))$, for storing the element, $O(1)$ for getting top element.

```cpp
struct TreeNode {
    int val;
    TreeNode* left;
    TreeNode* right;
    TreeNode()
        : val(0)
        , left(nullptr)
        , right(nullptr)
    {
    }
    TreeNode(int x)
        : val(x)
        , left(nullptr)
        , right(nullptr)
    {
    }
    TreeNode(int x, TreeNode* left, TreeNode* right)
        : val(x)
        , left(left)
        , right(right)
    {
    }
};

class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        if(root == nullptr) return {};

        queue<pair<TreeNode*, int>> bucket;

        bucket.push({ root, 1 });


        vector<int>tempBucket;
        vector<vector<int>>output;

        int previousLevel = 0;

        while (!bucket.empty()) {
            pair<TreeNode*, int> temp = bucket.front();

            if(temp.second != previousLevel){
                if(tempBucket.size())
                    output.push_back(tempBucket);
                tempBucket.clear();
                previousLevel = temp.second;
            }

            if (temp.first->left)
                bucket.push({ temp.first->left, temp.second + 1 });

            if (temp.first->right)
                bucket.push({ temp.first->right, temp.second + 1 });

            tempBucket.push_back(temp.first->val);

            bucket.pop();
        }

        if(tempBucket.size()) output.push_back(tempBucket);

        return output;
    }
};
```
